function normalizeCellValue(value?: string) {
  return typeof value === "string" ? value.trim() : "";

function parseEmployeesValue(value?: string) {
  const parsed = JSON.parse(normalizeCellValue(value) || "null") as {
    min: number;
    max: number;
  };

  return `${parsed.min.toString()}-${parsed.max.toString()}`;
}

): Promise<{ imported: number; skipped: number; errors: string[] }> {
  const session = await getSession();
  if (!session) throw new Error("Unauthorized");

  const file = formData.get("file") as File | null;
  if (!file) throw new Error("No file provided");

  const mappingRaw = formData.get("mapping") as string | null;
  const mapping: Record<string, string> = mappingRaw ? JSON.parse(mappingRaw) : {};

  const text = await file.text();
  const { data } = Papa.parse<Record<string, string>>(text, {
    header: true,
    skipEmptyLines: true,
  });

  const valid: any[] = [];
  const errors: string[] = [];

  data.forEach((rawRow, index) => {
    const row: Record<string, string> = {};
    if (Object.keys(mapping).length > 0) {
      Object.entries(mapping).forEach(([csvCol, targetField]) => {
        if (targetField && rawRow[csvCol] !== undefined) {
          row[targetField] = rawRow[csvCol];
        }
      });
    } else {
      Object.assign(row, rawRow);
    }

    const last_name = normalizeCellValue(row.last_name);
    const email = normalizeCellValue(row.email).toLowerCase();
    const mobile_phone = normalizeCellValue(row.mobile_phone);
    const company = normalizeCellValue(row.company);

      errors.push(`Row ${index + 2}: missing last_name or company`);
      return;

export async function importTargets(
  formData: FormData
): Promise<{ imported: number; skipped: number; errors: string[] }> {
  const session = await getSession();
  if (!session) throw new Error("Unauthorized");

  const file = formData.get("file") as File | null;
  if (!file) throw new Error("No file provided");

  const mappingRaw = formData.get("mapping") as string | null;
  const mapping: Record<string, string> = mappingRaw ? JSON.parse(mappingRaw) : {};

  const text = await file.text();
  const { data } = Papa.parse<Record<string, string>>(text, {
    header: true,
    skipEmptyLines: true,
  });

  const valid: any[] = [];
  const errors: string[] = [];

  data.forEach((rawRow, index) => {
    const row: Record<string, string> = {};
    if (Object.keys(mapping).length > 0) {
      Object.entries(mapping).forEach(([csvCol, targetField]) => {
        if (targetField && rawRow[csvCol] !== undefined) {
          row[targetField] = rawRow[csvCol];
        }
      });
    } else {
      Object.assign(row, rawRow);
    }

    const last_name = normalizeCellValue(row.last_name);
    const email = normalizeCellValue(row.email).toLowerCase();
    const mobile_phone = normalizeCellValue(row.mobile_phone);
    const company = row.company?.trim();
    const company_website = normalizeCellValue(row.company_website);
    const companyDomain = company_website || company.trim().toLowerCase();

    if (!last_name && !company) {
      errors.push(`Row ${index + 2}: missing last_name or company`);
      return;
    }

    valid.push({
      last_name: last_name ?? "",
      first_name: row.first_name || null,
      email: email || null,
      mobile_phone: mobile_phone || null,
      office_phone: row.office_phone || null,
      company: row.company || null,
      position: row.position || null,
      company_website: companyDomain || null,
      personal_website: row.personal_website || null,
      social_linkedin: row.social_linkedin || null,
      social_x: row.social_x || null,
      social_instagram: row.social_instagram || null,
      social_facebook: row.social_facebook || null,
      personal_email: row.personal_email || null,
      company_email: row.company_email || null,
      company_phone: row.company_phone || null,
      city: row.city || null,
      country: row.country || null,
      industry: row.industry || null,
      employees: row.employees || null,
      description: row.description || null,
      created_by: (session.user as any).id,
    });
  });

  if (valid.length > 0) {
    await prismadb.crm_Targets.createMany({ data: valid, skipDuplicates: true });
  }

  return { imported: valid.length, skipped: errors.length, errors };
}

Most likely root cause: the new `employees` parsing added in commit `9790f4c1` throws for blank, missing, or non-JSON employee values during CSV target import.

The risky path is in [actions/crm/targets/import-targets.ts](/home/dmytro/Github/phd-work/nextcrm-app-update/actions/crm/targets/import-targets.ts:11):

```ts
const parsed = JSON.parse(normalizeCellValue(value) || "null");
return `${parsed.min.toString()}-${parsed.max.toString()}`;
```

This is called unconditionally for every imported row at [line 59](/home/dmytro/Github/phd-work/nextcrm-app-update/actions/crm/targets/import-targets.ts:59), before the row is added to `valid`.

Failure cases:

- `employees` unmapped or absent: `JSON.parse("null")` returns `null`, then `parsed.min` throws.
- `employees` blank: same `null` path.
- legacy flat values like `1-10`, `10-50`, etc.: `JSON.parse(...)` throws a `SyntaxError`.
- malformed structured values: also throw.

This matches the incident evidence because the only changed backend file is the CSV import server action, the commit is specifically about “legacy employee range imports,” and the observed event is an unhandled request error rather than a handled per-row import validation error.

The modal also contributes to the miss: [components/modals/ImportTargetsModal.tsx](/home/dmytro/Github/phd-work/nextcrm-app-update/components/modals/ImportTargetsModal.tsx:155) only previews `last_name` / `company` validity and does not validate the `employees` shape, so a user can pass preview and fail only when the server action runs.

I do not think this is primarily a database/schema issue. `crm_Targets.employees` is an optional `String?`, and the failure happens before `createMany` is reached.

Recommended remediation: make `parseEmployeesValue` tolerant. It should return `null` for blank or missing input, accept existing string ranges where appropriate, validate JSON shape before dereferencing `min`/`max`, and convert bad employee values into row-level import errors instead of throwing an unhandled request error.

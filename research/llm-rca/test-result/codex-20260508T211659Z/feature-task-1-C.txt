**Why this is the strongest cause**

- The incident comes from `app.[locale].(routes).components.FeedbackForm` on `/en`.
- That component catches an Axios 500 and logs `"Feedback submission failed"`, exactly matching the incident.
- The server route has a deterministic 500 path for the payload currently sent by that component.
- The bug was introduced in commit `3fee71d` (`Refine feedback submission flow`) on 2026-05-01, one day before the incident on 2026-05-02. That commit changed the client to send `page`, while the server started reading `body.context.pathname`.

**Competing possibility**

`resendHelper()` can also return a 500 if the Resend API key is missing, and it runs before the route reads the request body. The incident payload does not include the `/api/feedback` response body, so this cannot be fully excluded from the provided evidence. However, the client/server payload mismatch is a clear recently introduced defect and would fail every authenticated feedback submission once Resend initialization succeeds.

**Secondary issue**

`FeedbackForm` always shows the success toast in `finally`, even after the catch path logs the failure. So affected users may see both an error and a “Thank you” success message.

**Recommended fix direction**

Align the payload contract: either send `context: { pathname: window.location.pathname }` from the client, or make the API read the existing `page` field. The API should also defensively validate the body before dereferencing nested fields.

import { NextResponse } from "next/server";

const logger = createLogger({ module: "app.api.client-logs" });
const levels = new Set(["debug", "info", "warn", "error"]);
const MAX_BODY_BYTES = 64 * 1024;

interface ClientLogBody {
  context?: unknown;
  level?: unknown;
  message?: unknown;
  pathname?: unknown;
  sessionId?: unknown;
  timestamp?: unknown;
  url?: unknown;
}

function limitString(value: unknown, fallback: string) {
  if (typeof value !== "string" || value.trim() === "") {
    return fallback;
  }

  return value.length > 1_000 ? `${value.slice(0, 1_000)}...[truncated]` : value;
}

export async function POST(req: Request) {
  const contentLength = Number(req.headers.get("content-length") ?? 0);
  if (contentLength > MAX_BODY_BYTES) {
    return NextResponse.json({ error: "Payload too large" }, { status: 413 });
  }

  let body: ClientLogBody;
  try {
    const parsed = await req.json();
    body = parsed && typeof parsed === "object" ? (parsed as ClientLogBody) : {};
  } catch {
    return NextResponse.json({ error: "Invalid JSON" }, { status: 400 });
  }

  const level = typeof body.level === "string" && levels.has(body.level) ? body.level : "info";
  const message = limitString(body?.message, "Client log");
  const payload = {
    client: {
      sessionId: limitString(body?.sessionId, "unknown"),
      pathname: limitString(body?.pathname, "unknown"),
      url: limitString(body?.url, "unknown"),
      timestamp: limitString(body?.timestamp, "unknown"),
      userAgent: limitString(req.headers.get("user-agent"), "unknown"),
    },
    context: body.context && typeof body.context === "object" ? body.context : {},
  };

  switch (level) {
    case "debug":
      logger.debug(payload, message);
      break;
    case "warn":
      logger.warn(payload, message);
      break;
    case "error":
      logger.error(payload, message);
      break;
    default:
      logger.info(payload, message);
  }

  return new NextResponse(null, { status: 204 });
}

const MAX_STRING_LENGTH = 4_000;
const MAX_DEPTH = 4;
const ENDPOINT = "/api/client-logs";

const sessionId =
  typeof crypto !== "undefined" && "randomUUID" in crypto
    ? crypto.randomUUID()
    : Math.random().toString(36).slice(2);

function truncate(value: string) {
  return value.length > MAX_STRING_LENGTH
    ? `${value.slice(0, MAX_STRING_LENGTH)}...[truncated]`
    : value;
}

function serialize(value: unknown, depth = 0, seen = new WeakSet<object>()): unknown {
  if (value instanceof Error) {
    return {
      name: value.name,
      message: truncate(value.message),
      stack: value.stack ? truncate(value.stack) : undefined,
    };
  }

  if (value === null || typeof value !== "object") {
    return typeof value === "string" ? truncate(value) : value;
  }

  if (seen.has(value)) {
    return "[Circular]";
  }

  if (depth >= MAX_DEPTH) {
    return "[MaxDepth]";
  }

  seen.add(value);

  if (Array.isArray(value)) {
    return value.slice(0, 50).map((item) => serialize(item, depth + 1, seen));
  }

  return Object.fromEntries(
    Object.entries(value as Record<string, unknown>)
      .slice(0, 50)
      .map(([key, item]) => [key, serialize(item, depth + 1, seen)])
  );
}

function send(level: ClientLogLevel, message: string, context: ClientLogContext) {
  if (typeof window === "undefined") {
    return;
  }

  const payload = JSON.stringify({
    level,
    message,
    context: serialize(context),
    sessionId,
    url: window.location.href,
    pathname: window.location.pathname,
    timestamp: new Date().toISOString(),
  });

  if (navigator.sendBeacon) {
    const sent = navigator.sendBeacon(
      ENDPOINT,
      new Blob([payload], { type: "application/json" })
    );
    if (sent) {
      return;
    }
  }

  fetch(ENDPOINT, {
    method: "POST",
    headers: { "content-type": "application/json" },
    body: payload,
    keepalive: true,
  }).catch(() => {
    // Client logging must never break the UI.
  });
}

export function createClientLogger(bindings: ClientLogContext) {
  const log = (level: ClientLogLevel, context: ClientLogContext, message: string) => {
    send(level, message, { ...bindings, ...context });
  };

  return {
    debug: (context: ClientLogContext, message: string) => log("debug", context, message),
    info: (context: ClientLogContext, message: string) => log("info", context, message),
    warn: (context: ClientLogContext, message: string) => log("warn", context, message),
    error: (context: ClientLogContext, message: string) => log("error", context, message),

**Incident Report**

Most likely root cause: a request-body contract mismatch between `FeedbackForm` and `/api/feedback`.

The client posts this payload from [FeedbackForm.tsx](/home/dmytro/Github/phd-work/nextcrm-app-update/app/[locale]/(routes)/components/FeedbackForm.tsx:48):

```ts
{
  ...data,
  page: window.location.pathname,
  submittedAt: new Date().toISOString()
}
```

But the API route reads [route.ts](/home/dmytro/Github/phd-work/nextcrm-app-update/app/api/feedback/route.ts:36):

```ts
const sourcePath = body.context.pathname.replace(/\/$/, "") || "/";
```

No `context` object is sent by the only `/api/feedback` caller in the repo. For a normal feedback submission, `body.context` is therefore `undefined`, so reading `.pathname` throws before the email-send `try/catch`. Next.js turns that unhandled server exception into a 500, which matches the Axios error logged by the client.

**Why this is the strongest cause**

- The incident comes from `app.[locale].(routes).components.FeedbackForm` on `/en`.
- That component catches an Axios 500 and logs `"Feedback submission failed"`, exactly matching the incident.
- The server route has a deterministic 500 path for the payload currently sent by that component.
- The bug was introduced in commit `3fee71d` (`Refine feedback submission flow`) on 2026-05-01, one day before the incident on 2026-05-02. That commit changed the client to send `page`, while the server started reading `body.context.pathname`.

**Competing possibility**

`resendHelper()` can also return a 500 if the Resend API key is missing, and it runs before the route reads the request body. The incident payload does not include the `/api/feedback` response body, so this cannot be fully excluded from the provided evidence. However, the client/server payload mismatch is a clear recently introduced defect and would fail every authenticated feedback submission once Resend initialization succeeds.

**Secondary issue**

`FeedbackForm` always shows the success toast in `finally`, even after the catch path logs the failure. So affected users may see both an error and a “Thank you” success message.

**Recommended fix direction**

Align the payload contract: either send `context: { pathname: window.location.pathname }` from the client, or make the API read the existing `page` field. The API should also defensively validate the body before dereferencing nested fields.
